19 /*----------------------------------------------------------------------*
20  * Button(pin, puEnable, invert, dbTime) instantiates a button object. *
21  * pin Is the Arduino pin the button is connected to. *
22  * puEnable Enables the AVR internal pullup resistor if != 0 (can also *
23  * use true or false). *
24  * invert If invert == 0, interprets a high state as pressed, low as *
25  * released. If invert != 0, interprets a high state as *
26  * released, low as pressed (can also use true or false). *
27  * dbTime Is the debounce time in milliseconds. *
28  * *
29  * (Note that invert cannot be implied from puEnable since an external *
30  * pullup could be used.) *
31  *----------------------------------------------------------------------*/
32 Button::Button(uint8_t pin, uint8_t puEnable, uint8_t invert, uint32_t dbTime)
33 {
34  _pin = pin;
35  _puEnable = puEnable;
36  _invert = invert;
37  _dbTime = dbTime;
38  if (_puEnable != 0)
39  pinMode(_pin, INPUT_PULLUP); //enable pullup resistor
40  else
41  pinMode(_pin, INPUT);
42  _state = digitalRead(_pin);
43  if (_invert != 0) _state = !_state;
44  _time = millis();
45  _lastState = _state;
46  _changed = 0;
47  _lastChange = _time;
48 }

50 /*----------------------------------------------------------------------*
51  * read() returns the state of the button, 1==pressed, 0==released, *
52  * does debouncing, captures and maintains times, previous states, etc. *
53  *----------------------------------------------------------------------*/
54 uint8_t Button::read(void)
55 {
56  static uint32_t ms;
57  static uint8_t pinVal;
58 
59  ms = millis();
60  pinVal = digitalRead(_pin);
61  if (_invert != 0) pinVal = !pinVal;
62  if (ms - _lastChange < _dbTime) {
63  _time = ms;
64  _changed = 0;
65  return _state;
66  }
67  else {
68  _lastState = _state;
69  _state = pinVal;
70  _time = ms;
71  if (_state != _lastState) {
72  _lastChange = ms;
73  _changed = 1;
74  }
75  else {
76  _changed = 0;
77  }
78  return _state;
79  }
80 }
