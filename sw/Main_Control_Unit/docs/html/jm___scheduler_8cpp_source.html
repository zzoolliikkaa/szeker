42 void jm_Scheduler::chain_insert()
43 {
44  jm_Scheduler *prev1 = 0;
45  jm_Scheduler *next1 = jm_Scheduler::first;
46 
47  while (next1 && jm_Scheduler_time_ge_time(this->time, next1->time))

49  prev1 = next1;

52 
53  if (prev1) prev1->next = this; else jm_Scheduler::first = this;

106 {

108  jm_Scheduler *wakeup_next1 = jm_Scheduler::wakeup_first;
109 
110  while (wakeup_next1) // loop wakeup_chain
111  {
112  if (wakeup_next1 == this) // this found in wakeup_chain ?
113  {
114  // remove from wakeup_chain

116  wakeup_prev1->wakeup_next = this->wakeup_next;

119  this->wakeup_next = 0;

121  return; // now removed from wakeup_chain, exit function
122  }
123 
124  wakeup_prev1 = wakeup_next1;
125  wakeup_next1 = wakeup_next1->wakeup_next;

230 void jm_Scheduler::cycle()
231 {
232  for (;;)
233 {
234  jm_Scheduler::time_cycle();
235 
236  cli();
237 
238  if (jm_Scheduler::wakeup_first)
239  {
240  jm_Scheduler *wakeup_first0 = jm_Scheduler::wakeup_first;
241 
242  // remove wakeup_first
243  jm_Scheduler::wakeup_first = wakeup_first0->wakeup_next;
244  wakeup_first0->wakeup_next = 0;
245 
246  // remove wakeuped routine from routine chain
247 // jm_Scheduler::wakeup_first->chain_remove();
248  wakeup_first0->chain_remove();
249 
250  // insert wakeuped routine at first routine
251  wakeup_first0->next = jm_Scheduler::first;
252  jm_Scheduler::first = wakeup_first0;
253 
254  sei();
255  }
256  else
257  {
258  sei();
259 
260  if (jm_Scheduler::first == 0) break;
261 
262  if (!jm_Scheduler_tref_ge_time(jm_Scheduler::first->time)) break;
263  }
264 
265  // set crnt with first
266  jm_Scheduler::crnt = jm_Scheduler::first;
267 
268  // remove first from chain routine
269  jm_Scheduler::first = jm_Scheduler::crnt->next;
270  jm_Scheduler::crnt->next = 0;
271 
272  jm_Scheduler::crnt->func(); // call routine function
273 
274  if (jm_Scheduler::crnt->stopping || jm_Scheduler::crnt->ival == 0) // stopping or not rearmed ?
275  {
276  jm_Scheduler::crnt->started = false;
277  jm_Scheduler::crnt->stopping = false;
278 
279  jm_Scheduler::crnt->wakeup_chain_remove();
280  }
281  else // rearmed/intervalled calls
282  {
283  jm_Scheduler::crnt->time += jm_Scheduler::crnt->ival; // synchonous
284 
285  jm_Scheduler::crnt->chain_insert();
286  }
287 
288  jm_Scheduler::crnt = 0;
289 }
